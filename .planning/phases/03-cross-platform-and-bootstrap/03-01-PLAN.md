2. Enhance `src/core/path-rewriter.ts` `rewritePathsForRepo`:
   - After replacing `homeDir` with `{{HOME}}`, also replace the forward-slash version of homeDir (in case content has mixed separators on Windows)
   - After all replacements, normalize any remaining backslash path separators adjacent to `{{HOME}}` to forward slashes. Specifically: replace `{{HOME}}\\` with `{{HOME}}/` to ensure stored paths use POSIX separators.
   - Import `path` from `node:path` for `path.sep` check.

```typescript
import * as path from "node:path";

export function rewritePathsForRepo(content: string, homeDir: string): string {
  let result = content.replaceAll(homeDir, "{{HOME}}");
  // On Windows, homeDir uses backslashes. Also handle forward-slash variant
  if (path.sep === "\\") {
    const forwardSlashHome = homeDir.replaceAll("\\", "/");
    result = result.replaceAll(forwardSlashHome, "{{HOME}}");
  }
  // Normalize any remaining backslash separators after {{HOME}} token
  result = result.replaceAll("{{HOME}}\\", "{{HOME}}/");
  return result;
}
```
